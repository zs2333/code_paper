## 运算符

### 数据的输入

关键字 cin

```C++
#include <iostream>
#include <string>//用C++风格的字符串时，需要包含对应的头文件
using namespace std;
int main()
{
	//整型
	int a = 0;
	cout << "请给变量a赋值 " << endl;
	cin >> a;
	cout << "整型变量a " << a << endl;
	//浮点型
	float b = 3.14f;
	cout << "请给变量b赋值 " << endl;
	cin >> b;
	cout << "浮点型变量f的值是： " << b << endl;
	//字符型
	string c = "";
	cout << "请给字符串c赋值" << endl;
	cin >> c;
	cout << "字符串c的值是： " << c << endl;

	system("pause");

	return 0;
}
```
